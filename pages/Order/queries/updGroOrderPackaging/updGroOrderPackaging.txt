update public.gro_order
set
  packaging_id = {{
    appsmith.store.current_packaging_id === "" ||
    appsmith.store.current_packaging_id == null
      ? null
      : Number(appsmith.store.current_packaging_id)
  }},
  updated_at = now()
where id = {{ Number(appsmith.store.current_gro_order_id || 0) }}
  and client_id = '{{ appsmith.store.current_client_id }}'::uuid
returning id, packaging_id;
