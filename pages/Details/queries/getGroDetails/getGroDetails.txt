select
  line_no,
  event_datetime_display,
  event_ref,
  event_name,
  ingredient_name,
  category_name,
  supplier_name,
  required_qty,
  required_unit
from public.gro_details
where batch_id = 1
  and client_id = '{{ appsmith.store.current_client_id }}'::uuid
order by
  lower(event_name),
  line_no;
