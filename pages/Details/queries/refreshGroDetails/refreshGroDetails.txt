delete from public.gro_details
where batch_id = 1
  and client_id = '{{ appsmith.store.current_client_id }}'::uuid;

insert into public.gro_details
(
  client_id,
  batch_id,
  gro_event_id,
  event_id,
  line_no,
  event_ref,
  event_datetime,
  event_datetime_display,
  event_name,
  ingredient_id,
  ingredient_name,
  category_name,
  supplier_name,
  required_qty,
  required_unit_id,
  required_unit
)
select
  '{{ appsmith.store.current_client_id }}'::uuid,
  1,
  ge.gro_event_id,
  gei.event_id,
  row_number() over (
    partition by gei.event_id
    order by lower(gei.ingredient_name)
  ),
  gei.event_ref,
  gei.event_datetime,
  gei.event_datetime_display,
  gei.event_name,
  gei.ingredient_id,
  gei.ingredient_name,
  gei.category_name,
  gei.supplier_name,
  round(gei.needed_qty, 2),
  gei.needed_unit_id,
  gei.needed_unit
from public.vw_gro_event_ingredients gei
join public.vw_gro_events ge
  on ge.event_id = gei.event_id
where ge.client_id = '{{ appsmith.store.current_client_id }}'::uuid
  and ge.batch_id = 1
  and ge.to_order = true
order by
  lower(gei.event_name),
  lower(gei.ingredient_name);

select count(*) as detail_rows
from public.gro_details
where batch_id = 1
  and client_id = '{{ appsmith.store.current_client_id }}'::uuid;
