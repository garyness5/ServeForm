select
  name
from public.ing_items
where client_id = {{ appsmith.store.current_client_id }}
  and deleted = false
  and (
    name = {{ appsmith.store.IngDuplicate_base_name + ' - Copy' }}
    or name ~ {{ '^' + appsmith.store.IngDuplicate_base_name.replace(/[.*+?^${}()|[\]\\]/g, '\\$&') + ' - Copy [0-9]+$' }}
  )
order by name;
