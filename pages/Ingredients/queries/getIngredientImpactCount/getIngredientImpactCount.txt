select
  (
    select count(*)
    from public.rec_components
    where item_type = 'ingredient'
      and ingredient_id = {{ Number(tblIngList.selectedRow?.id || 0) }}
  ) as recipe_count,

  (
    select count(*)
    from public.dsh_components
    where item_type = 'ingredient'
      and ingredient_id = {{ Number(tblIngList.selectedRow?.id || 0) }}
  ) as dish_count,

  0 as menu_count,
  0 as event_count;
