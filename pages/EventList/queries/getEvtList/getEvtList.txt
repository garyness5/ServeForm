select
  id,
  event_date,
  event_time,
  event_datetime,
  event_datetime_display,
  event_ref,
  name,
  customer_name,
  status,
  active,
  menu_count,
  total_guests,
  to_produce,
  total_cost,
  warning_count,
  deleted
from public.vw_evt_items
where client_id = '{{ appsmith.store.current_client_id }}'::uuid
  and coalesce(deleted, false) = false
order by event_datetime nulls last, lower(name), id;
