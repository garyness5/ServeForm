update public.evt_items
set
  status = '{{ (tblEvtList.updatedRow.status || tblEvtList.triggeredRow.status || "Draft").replaceAll("'", "''") }}',
  active = {{ tblEvtList.updatedRow.active === false ? false : true }},
  updated_at = now()
where id = {{ Number(tblEvtList.updatedRow.id || tblEvtList.triggeredRow.id || 0) }}
  and client_id = '{{ appsmith.store.current_client_id }}'::uuid
  and coalesce(deleted, false) = false
returning id, status, active;
