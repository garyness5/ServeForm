with excluded_generated as (
  select distinct
    gd.event_id
  from public.gro_details gd
  join public.gro_events ge
    on ge.event_id = gd.event_id
   and ge.client_id = gd.client_id
   and ge.batch_id = gd.batch_id
  where gd.client_id = '{{ appsmith.store.current_client_id }}'::uuid
    and gd.batch_id = 1
    and coalesce(ge.to_order, false) = false
),
manual_values as (
  select exists (
    select 1
    from public.gro_order go
    where go.client_id = '{{ appsmith.store.current_client_id }}'::uuid
      and go.batch_id = 1
      and (
        go.buy_qty is not null
        or go.packs_to_order is not null
      )
  ) as has_manual_values
),
participating_events as (
  select count(*) as participating_count
  from public.gro_events ge
  where ge.client_id = '{{ appsmith.store.current_client_id }}'::uuid
    and ge.batch_id = 1
    and coalesce(ge.to_order, false) = true
)
select
  (select count(*) from excluded_generated) as excluded_generated_count,
  (select has_manual_values from manual_values) as has_manual_values,
  (select participating_count from participating_events) as participating_count;
