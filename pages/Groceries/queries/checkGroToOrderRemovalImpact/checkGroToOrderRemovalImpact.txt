select
  count(distinct gd.event_id) as excluded_generated_count,

  exists (
    select 1
    from public.gro_order go
    where go.client_id = '{{ appsmith.store.current_client_id }}'::uuid
      and go.batch_id = 1
      and (
        go.buy_qty is not null
        or go.packs_to_order is not null
      )
  ) as has_manual_values,

  (
    select count(*)
    from public.gro_events ge2
    where ge2.client_id = '{{ appsmith.store.current_client_id }}'::uuid
      and ge2.batch_id = 1
      and coalesce(ge2.to_order, false) = true
  ) as participating_count

from public.gro_details gd
join public.gro_events ge
  on ge.client_id = gd.client_id
 and ge.batch_id = gd.batch_id
 and ge.event_id = gd.event_id
where gd.client_id = '{{ appsmith.store.current_client_id }}'::uuid
  and gd.batch_id = 1
  and coalesce(ge.to_order, false) = false;
