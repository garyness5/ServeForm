select
  gro_event_id,
  batch_id,
  event_id,
  event_ref,
  event_datetime,
  event_datetime_display,
  event_name,
  total_guests,
  to_order,
  notes
from public.vw_gro_events
where client_id = '{{ appsmith.store.current_client_id }}'::uuid
  and event_status = 'Ordered'
  and event_active = true
  and coalesce(event_deleted, false) = false
order by event_datetime nulls last, lower(event_name);
