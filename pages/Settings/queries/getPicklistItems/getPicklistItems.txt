SELECT
  hli.id,
  hli.list_id,
  hli.name,
  hli.sort_order,
  hli.is_system
FROM helper_list_items hli
JOIN helper_lists hl
  ON hl.id = hli.list_id
WHERE hli.list_id = {{ Number(selPicklistType.selectedOptionValue) }}
  AND (
    (
      hl.list_code = 'diet_tag'
      AND (
        hli.is_system = true
        OR hli.client_id = '{{ appsmith.store.current_client_id }}'::uuid
      )
    )
    OR
    (
      hl.list_code <> 'diet_tag'
      AND hli.client_id = '{{ appsmith.store.current_client_id }}'::uuid
    )
  )
ORDER BY hli.sort_order NULLS LAST, hli.name;
