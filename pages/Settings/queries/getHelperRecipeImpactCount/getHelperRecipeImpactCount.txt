select
  case
    when (
      select list_code
      from helper_lists
      where id = {{ Number(selPicklistType.selectedOptionValue) }}
    ) = 'recipe_category'
      then (
        select count(*)
        from rec_items
        where client_id = '{{ appsmith.store.current_client_id }}'::uuid
          and category_id = {{ Number(tblPicklistItems.selectedRow.id) }}
          and coalesce(deleted, false) = false
      )

    when (
      select list_code
      from helper_lists
      where id = {{ Number(selPicklistType.selectedOptionValue) }}
    ) = 'diet_tag'
      then (
        select count(distinct rdt.recipe_id)
        from rec_diet_tags rdt
        join rec_items r
          on r.id = rdt.recipe_id
        where r.client_id = '{{ appsmith.store.current_client_id }}'::uuid
          and rdt.helper_list_item_id = {{ Number(tblPicklistItems.selectedRow.id) }}
          and coalesce(r.deleted, false) = false
      )

    else 0
  end as recipe_count;
