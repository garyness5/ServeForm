with recipe_costs as (
  select
    vc.recipe_id,
    sum(coalesce(vc.line_cost, 0)) as subtotal
  from public.vw_rec_components vc
  group by vc.recipe_id
)

select
  'ingredient'::text as item_type,
  id,
  name,
  category_name,
  yield_unit_id as default_unit_id,
  yield_unit as default_unit,
  unit_type,
  wastage_percent,
  allergen_names,
  diet_tag_names,
  price_per_unit,
  cost_per_base_unit,
  factor_to_base
from public.vw_ing_items
where client_id = '{{ appsmith.store.current_client_id }}'::uuid
  and active = true

union all

select
  'recipe'::text as item_type,
  r.id,
  r.name,
  r.category_name,
  r.yield_unit_id as default_unit_id,
  r.yield_unit as default_unit,
  r.yield_unit_type as unit_type,
  0::numeric as wastage_percent,
  null::text as allergen_names,
  r.diet_tag_names,
  case
    when r.yield_qty > 0
    then (coalesce(rc.subtotal, 0) * (1 + coalesce(r.extra_percent, 0) / 100)) / r.yield_qty
    else null
  end as price_per_unit,
  case
    when r.yield_qty > 0 and u.factor_to_base > 0
    then (coalesce(rc.subtotal, 0) * (1 + coalesce(r.extra_percent, 0) / 100)) / (r.yield_qty * u.factor_to_base)
    else null
  end as cost_per_base_unit,
  u.factor_to_base
from public.vw_rec_items r
left join public.units u
  on u.id = r.yield_unit_id
left join recipe_costs rc
  on rc.recipe_id = r.id
where r.client_id = '{{ appsmith.store.current_client_id }}'::uuid
  and r.active = true
  and coalesce(r.deleted, false) = false
  and r.id <> {{ Number(appsmith.store.current_recipe_id || 0) }}

order by item_type, name;
