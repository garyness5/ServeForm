update public.quo_events
set
  contact_id = nullif(
    {{ Number(selQuoContact.selectedOptionValue || 0) }},
    0
  )::bigint,

  customer_id = case
    when {{ Number(selQuoContact.selectedOptionValue || 0) }} > 0 then (
      select customer_id
      from public.customer_contacts
      where id = {{ Number(selQuoContact.selectedOptionValue || 0) }}::bigint
    )
    else customer_id
  end,

  updated_at = now()
where id = {{ Number(appsmith.store.selectedQuoteEventId || 0) }}::bigint
  and client_id = '{{ appsmith.store.current_client_id }}'::uuid
  and coalesce(deleted, false) = false
returning id, customer_id, contact_id;
