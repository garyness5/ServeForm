select
  id,
  name,
  category_name,
  serve_form_name,
  yield_qty,
  yield_unit,
  total_cost,
  cost_per_unit,
  active,
  diet_tag_names,
  deleted
from public.vw_dsh_items
where client_id = '{{ appsmith.store.current_client_id }}'::uuid
  and coalesce(deleted, false) = false
order by lower(name);
